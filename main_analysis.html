# =========================================================

print("\n========== CONFIDENCE INTERVAL ==========")

n = len(y_test)

p_hat = rf_accuracy

margin_error = 1.96 * np.sqrt((p_hat * (1 - p_hat)) / n)

lower = p_hat - margin_error

upper = p_hat + margin_error

print("Accuracy:", p_hat)

print("95% Confidence Interval:")

print(lower, upper)

# 16. HİPOTEZ TESTİ
# =========================================================

print("\n========== HYPOTHESIS TEST ==========")

# H0: Model accuracy <= 0.5
# H1: Model accuracy > 0.5

p0 = 0.5

z_score = (p_hat - p0) / np.sqrt((p0 * (1 - p0)) / n)

p_value = 1 - norm.cdf(z_score)

print("Z-score:", z_score)

print("P-value:", p_value)

if p_value < 0.05:
    print("H0 reddedilir.")
    print("Model istatistiksel olarak anlamlıdır.")
else:
    print("H0 reddedilemez.")
